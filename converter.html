    <input type="number" id="inputValue" placeholder="Enter a number">

    <div class="toggle-container">

      <span>Km → Mi</span>

      <label class="switch">

        <input type="checkbox" id="modeToggle">

        <span class="slider"></span>

      </label>

      <span>Mi → Km</span>

    </div>


    <button onclick="convert()">Convert</button>

    <div class="result" id="result">Result will appear here</div>

  </div>


  <script>

    function convert() {

      const value = parseFloat(document.getElementById("inputValue").value);

      const isMilesToKm = document.getElementById("modeToggle").checked;

      const resultBox = document.getElementById("result");


      if (isNaN(value)) {

        resultBox.textContent = "⚠️ Please enter a valid number.";

        return;

      }


      if (isMilesToKm) {

        // Miles → Kilometers

        const km = value / 0.621371;

        resultBox.textContent = `${value} miles = ${km.toFixed(2)} km`;

      } else {

        // Kilometers → Miles

        const miles = value * 0.621371;

        resultBox.textContent = `${value} km = ${miles.toFixed(2)} miles`;

      }

    }

  </script>
